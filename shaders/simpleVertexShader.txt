#version 330

in vec3 position;
in vec3 normals;

out vec3 n_eye;
out vec3 position_eye, normal_eye;

uniform mat4 view;
uniform mat4 proj;
uniform mat4 model;

void main(){
	n_eye = (view * vec4 (normals, 0.0)).xyz;
	position_eye = vec3 (view * model * vec4 (position, 1.0));
	normal_eye = vec3 (view * model * vec4 (normals, 0.0));
	gl_Position =  proj * view * model * vec4 (position, 1.0);
}
